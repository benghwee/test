@Service
public class MitmProxyService {

    private HttpProxyServer proxyServer;

    @PostConstruct
    public void startProxy() throws Exception {
        // Load your PFX for upstream (client) auth
        KeyStore pfxStore = KeyStore.getInstance("PKCS12");
        try (InputStream in = new FileInputStream("path/to/client.pfx")) {
            pfxStore.load(in, "password".toCharArray());
        }

        // Configure MITM Manager to intercept traffic
        SelfSignedMitmManager mitmManager = SelfSignedMitmManager.builder()
            .build(); // In a real MITM, this generates/uses a CA to sign on-the-fly certs

        proxyServer = DefaultHttpProxyServer.bootstrap()
            .withPort(8081)
            .withManInTheMiddle(mitmManager)
            .withFiltersSource(new HttpFiltersSourceAdapter() {
                @Override
                public HttpFilters filterRequest(HttpRequest originalRequest) {
                    return new HttpFiltersAdapter(originalRequest) {
                        @Override
                        public HttpResponse clientToProxyRequest(HttpObject httpObject) {
                            // INTERCEPT: Log or modify request here
                            System.out.println("Intercepted: " + originalRequest.uri());
                            return null;
                        }
                    };
                }
            })
            .start();
    }
}
